# GCD
- lcm要先求gcd $$gcd(x, y) * lcm(x, y) = x * y$$
- 多个数gcd,lcm可以递归算出

```python
def gcd(x, y):
    while y != 0:
        x, y = y, x % y
    return x

def lcm(x, y):
    return x * y // gcd(x, y)
```

# 快速幂

```python
def binpow(a, b, mod=None):
    res = 1
    while b > 0:
        if (b & 1):
            res = ((res * a) % mod) if mod else (res * a)
        a = ((a * a) % mod) if mod else (a * a)
        b >>= 1
    return res
```

# 质因子分解

```python
def breakdown(n):
    result = []
    for i in range(2, math.isqrt(n) + 1):
        if n % i == 0: # 如果 i 能够整除 N，说明 i 为 n 的一个质因子。
            while n % i == 0:
                n //= i
            result.append(i)
    if n != 1: # 说明再经过操作之后 n 留下了一个素数
        result.append(n)
    return result
```

# 素数筛

## 埃拉托斯特尼筛 Sieve of Eratosthenes

```python
def prime(n):
    flag = [True] * (n + 1) # 加一个用不着的0
    flag[0] = flag[1] = False # 方便使用sum计算质数个数
    for i in range(2, math.isqrt(n)+1):
        if flag[i]:
            for j in range(i*i, n+1, i):
                flag[j] = False
    return [i for i, isprime in enumerate(flag) if isprime]
```

```python
    prime = []

            prime.append(i)

                """
```
